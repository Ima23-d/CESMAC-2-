void soma(int n_1, int n_2){

}

    printf("SUBTRAÇÂO\n");
    printf("%d - %d = %d\n", n_1,n_2,n_1-n_2);
}
void divisao(int n_1, int n_2){
    printf("Divisão\n");
    printf("%d / %d = %d\n", n_1,n_2,n_1/n_2);
}
void multiplicacao(int n_1, int n_2){
    printf("Multiplicação\n");
    printf("%d * %d = %d\n", n_1,n_2,n_1*n_2);
