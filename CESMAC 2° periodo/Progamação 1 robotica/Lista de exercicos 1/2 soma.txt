
#include <stdio.h>
void soma(int n1,int n2){
    printf("%d + %d = %d",n1,n2,n1+n2);
}
int main()
{   int n1,n2;
    printf("Digite o valor de n1: \n");
    scanf("%d",&n1);
    printf("Digite o valor de n2: \n");
    scanf("%d",&n2);
    while (n1 <=0 || n2 <=0){
         printf("Numero invalido!\n");
        printf("Digite o valor de n1: \n");
        scanf("%d",&n1);
        printf("Digite o valor de n2: \n");
        scanf("%d",&n2);
    }
    soma(n1,n2);

    return 0;
}
