#include <stdio.h>
#include <locale.h>
void maior(int n1, int n2){
    if (n1 > n2){
        printf("%d é maior que %d",n1,n2);
    }else if (n2 > n1){
         printf("%d é maior que %d",n2,n1);
    }else{
        printf("os numeros são iguais");
    }
}
int main() {
    setlocale(LC_ALL, "");
    int n1,n2;
    printf("Digite o n1 \n");
    scanf("%d",&n1);
    printf("Digite o n2 \n");
    scanf("%d",&n2);
    while (n1 <=0 || n2 <=0){
        printf("Números inválidos\n");
        printf("Digite o n1 \n");
        scanf("%d",&n1);
        printf("Digite o n2 \n");
        scanf("%d",&n2);
    }
    maior(n1,n2);
    return 0;
}
