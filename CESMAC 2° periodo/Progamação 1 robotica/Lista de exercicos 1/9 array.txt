#include <stdio.h>
#include <locale.h>

void maiorElemento(int numeros[], int tamanho) {
    int maior = numeros[0]; 
    for (int i = 1; i < tamanho; i++) {
        if (numeros[i] > maior) {
            maior = numeros[i];
        }
    }
    printf("O maior elemento Ã©: %d\n", maior);
}

int main() {
    setlocale(LC_ALL, "");

    int numeros[] = {5, 8, 12, 3, 15, 7};
    int tamanho = sizeof(numeros) / sizeof(numeros[0]);

    maiorElemento(numeros, tamanho);

    return 0;
}
